	function countDown(seconds, fn, obj, type, issue) {
		var timer = setInterval(function() {
			if (seconds >= 0) {
				msg = seconds;
				fn(msg);
				--seconds;
			} else {
				clearInterval(timer);
				obj.parent(".countdown").html('开奖中...');
				setTimeout(function(){getNew(obj, type, issue)}, 2000);
			}
		}, 1000);
	}

	function getNew(obj, type, issue) {
		$.post("/api.php/Index/isOpen",	{type: type, issue: issue},function(data){
			if(data.data){
				location.reload();
			}else{
				countDown(0, function(msg){obj.parent(".countdown").html('开奖中...');}, obj, type, issue);
			}
		})
	}
</script>
